 interface Playable {
    void play();
    void pause();
}

class MusicPlayer {
    public void play() {
        System.out.println("Playing music");
    }
    public void pause() {
        System.out.println("Music paused");
    }
}

public class Main {
    public static void main(String[] args) {
        Playable player = new MusicPlayer();
        player.play();
    }
}
