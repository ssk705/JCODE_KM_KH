public class ArrayStack {
    private int[] stack;
    private int top;
    private int capacity;
    
    public ArrayStack(int size) {
        stack = new int[size];
        capacity = size;
        top = -1;
    }
    
    public void push(int value) {
        stack[++top] = value;
    }
}
