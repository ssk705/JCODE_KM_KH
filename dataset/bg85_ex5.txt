public class LinkedList {
    Node head;
    class Node {
        int data;
        Node next;
        Node(int d) { data = d; }
    }
    int getSum() {
        Node current = head;
        int sum = 0;
        do {
            sum += current.data;
            current = current.next;
        } while (current != null);
        return sum;
    }
}
